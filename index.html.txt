    <script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js" integrity="sha256-20nQCchB9co0qIjJZRGuk2/Z9VM+kNiyxNV1lvTlZBo=" crossorigin=""></script>
    <script>
        // CONFIGURACIÓN CONSTANTE
        const SECONDS_PER_DELIVERY = 300; // 5 minutos por paquete
        
        // ESTADO DE LA APLICACIÓN
        let map;
        let stops = []; // Array de objetos: { id, name, display_name, lat, lon, marker }
        let currentRouteLayer = null; // Capa de la polilínea (la línea azul en el mapa)
        let searchTimeout = null;

        // ELEMENTOS DEL DOM
        const inputElement = document.getElementById('addressInput');
        const searchResultsList = document.getElementById('searchResults');
        const searchLoading = document.getElementById('searchLoading');
        const listElement = document.getElementById('stopList');
        const emptyState = document.getElementById('emptyState');
        const countElement = document.getElementById('stopCount');
        const calcManualBtn = document.getElementById('calcManualBtn');
        const optimizeBtn = document.getElementById('optimizeBtn');
        const exportGoogleMapsBtn = document.getElementById('exportGoogleMapsBtn');
        const clearBtn = document.getElementById('clearBtn');
        const statsPanel = document.getElementById('statsPanel');
        const loadingOverlay = document.getElementById('loadingOverlay');

        // INICIALIZACIÓN DEL MAPA LEAFLET
        function initMap() {
            // Centrar en un punto global por defecto (ej. centro de España/Europa o EEUU según preferencia)
            // Aquí usamos un zoom alejado
            map = L.map('map', {
                zoomControl: false // Lo añadiremos abajo a la derecha
            }).setView([40.4168, -3.7038], 5); // Por defecto centrado en España

            L.control.zoom({ position: 'bottomright' }).addTo(map);

            // Añadir la capa de "azulejos" (tiles) de OpenStreetMap
            L.tileLayer('https://{s}.basemaps.cartocdn.com/rastertiles/voyager/{z}/{x}/{y}{r}.png', {
                attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors &copy; <a href="https://carto.com/attributions">CARTO</a>',
                subdomains: 'abcd',
                maxZoom: 20
            }).addTo(map);

            // Intentar obtener la ubicación del usuario
            if ("geolocation" in navigator) {
                navigator.geolocation.getCurrentPosition(position => {
                    if (stops.length === 0) { // Solo si no ha añadido paradas ya
                        map.setView([position.coords.latitude, position.coords.longitude], 12);
                    }
                }, () => { console.log("Geolocalización no disponible."); });
            }
        }

        // BÚSQUEDA DE DIRECCIONES (GEOCORDING INVERSO CON NOMINATIM)
        inputElement.addEventListener('input', (e) => {
            const query = e.target.value.trim();
            clearTimeout(searchTimeout);
            
            if (query.length < 3) {
                searchResultsList.classList.add('hidden');
                searchLoading.classList.add('hidden');
                return;
            }

            searchLoading.classList.remove('hidden');

            // Debounce: Esperar 600ms después de que el usuario deje de teclear para no saturar la API
            searchTimeout = setTimeout(async () => {
                try {
                    // API Nominatim (OpenStreetMap Search)
                    const response = await fetch(`https://nominatim.openstreetmap.org/search?format=json&q=${encodeURIComponent(query)}&addressdetails=1&limit=5`, {
                        headers: {
                            'Accept-Language': 'es' // Pedir resultados en español
                        }
                    });
                    
                    if (!response.ok) throw new Error("Error en la red");
                    
                    const data = await response.json();
                    renderSearchResults(data);
                } catch (error) {
                    console.error("Error buscando:", error);
                    showToast("Error conectando con el buscador de direcciones.", "error");
                } finally {
                    searchLoading.classList.add('hidden');
                }
            }, 600);
        });

        // Ocultar resultados al hacer click fuera
        document.addEventListener('click', (e) => {
            if (e.target !== inputElement && !searchResultsList.contains(e.target)) {
                searchResultsList.classList.add('hidden');
            }
        });

        function renderSearchResults(results) {
            searchResultsList.innerHTML = '';
            
            if (results.length === 0) {
                searchResultsList.innerHTML = '<li class="p-3 text-sm text-slate-500 text-center">No se encontraron resultados</li>';
                searchResultsList.classList.remove('hidden');
                return;
            }

            results.forEach(place => {
                const li = document.createElement('li');
                li.className = 'p-3 hover:bg-emerald-50 cursor-pointer border-b border-slate-100 last:border-b-0 transition-colors';
                
                // Extraer un nombre principal corto
                const title = place.address.road || place.address.pedestrian || place.address.village || place.name || "Ubicación";
                
                li.innerHTML = `
                    <div class="font-medium text-slate-800 text-sm truncate">${title}</div>
                    <div class="text-xs text-slate-500 truncate mt-0.5">${place.display_name}</div>
                `;
                
                li.addEventListener('click', () => {
                    addStop({
                        id: place.place_id || Date.now(),
                        name: title,
                        display_name: place.display_name,
                        lat: parseFloat(place.lat),
                        lon: parseFloat(place.lon)
                    });
                    inputElement.value = '';
                    searchResultsList.classList.add('hidden');
                });
                
                searchResultsList.appendChild(li);
            });
            
            searchResultsList.classList.remove('hidden');
        }

        // GESTIÓN DE PARADAS (STOPS)
        function addStop(placeData) {
            if (stops.length >= 25) {
                showToast("Límite máximo de paradas alcanzado (25).", "warning");
                return;
            }

            // Crear icono personalizado para Leaflet
            const isOrigin = stops.length === 0;
            const index = stops.length + 1;
            const color = isOrigin ? '#dc2626' : '#059669'; // Rojo para origen, Esmeralda para el resto
            
            const markerHtml = `
                <div style="background-color: ${color}; width: 28px; height: 28px; border-radius: 50%; display: flex; align-items: center; justify-content: center; color: white; font-weight: bold; border: 2px solid white; box-shadow: 0 2px 5px rgba(0,0,0,0.3); font-size: 12px;">
                    ${index}
                </div>
            `;
            
            const customIcon = L.divIcon({
                className: 'custom-div-icon',
                html: markerHtml,
                iconSize: [28, 28],
                iconAnchor: [14, 14], // Centrar sobre el punto exacto
                popupAnchor: [0, -14]
            });

            // Añadir marcador al mapa
            const marker = L.marker([placeData.lat, placeData.lon], { icon: customIcon })
                .addTo(map)
                .bindPopup(`<b>${isOrigin ? 'Origen' : 'Parada ' + index}</b><br>${placeData.name}`);

            // Guardar en nuestro array
            const newStop = { ...placeData, marker: marker };
            stops.push(newStop);

            updateUI();
            adjustMapBounds();
        }

        function removeStop(index) {
            // Remover marcador del mapa
            map.removeLayer(stops[index].marker);
            
            // Remover del array
            stops.splice(index, 1);
            
            // Limpiar la ruta dibujada si existe porque los puntos cambiaron
            clearRoute();
            
            // Actualizar números de los marcadores restantes
            refreshMarkerIcons();
            
            updateUI();
            if(stops.length > 0) adjustMapBounds();
        }

        function refreshMarkerIcons() {
            stops.forEach((stop, i) => {
                const isOrigin = i === 0;
                const color = isOrigin ? '#dc2626' : '#059669';
                
                const markerHtml = `
                    <div style="background-color: ${color}; width: 28px; height: 28px; border-radius: 50%; display: flex; align-items: center; justify-content: center; color: white; font-weight: bold; border: 2px solid white; box-shadow: 0 2px 5px rgba(0,0,0,0.3); font-size: 12px;">
                        ${i + 1}
                    </div>
                `;
                
                stop.marker.setIcon(L.divIcon({
                    className: 'custom-div-icon',
                    html: markerHtml,
                    iconSize: [28, 28],
                    iconAnchor: [14, 14],
                    popupAnchor: [0, -14]
                }));
                stop.marker.setPopupContent(`<b>${isOrigin ? 'Origen' : 'Parada ' + (i + 1)}</b><br>${stop.name}`);
            });
        }

        function adjustMapBounds() {
            if (stops.length === 0) return;
            if (stops.length === 1) {
                map.setView([stops[0].lat, stops[0].lon], 15);
                return;
            }
            // Crear un grupo de características para obtener los límites
            const group = new L.featureGroup(stops.map(s => s.marker));
            // Animar el zoom para ajustar todos los marcadores (con un poco de padding)
            map.fitBounds(group.getBounds(), { padding: [50, 50], animate: true, duration: 0.5 });
        }

        function updateUI() {
            // Actualizar contador
            countElement.textContent = stops.length;
            
            const canCalculate = stops.length >= 2;
            calcManualBtn.disabled = !canCalculate;
            optimizeBtn.disabled = !canCalculate;
            exportGoogleMapsBtn.disabled = !canCalculate;

            // Renderizar lista HTML
            listElement.innerHTML = '';
            
            if (stops.length === 0) {
                listElement.appendChild(emptyState);
                emptyState.style.display = 'flex';
                statsPanel.classList.remove('scale-100');
                statsPanel.classList.add('scale-0');
                return;
            }
            
            emptyState.style.display = 'none';

            stops.forEach((stop, index) => {
                const li = document.createElement('li');
                li.className = 'flex justify-between items-start p-3 bg-white border border-slate-200 rounded-lg shadow-sm hover:shadow-md transition-shadow group';
                
                const badge = index === 0 
                    ? '<span class="ml-2 inline-flex items-center px-1.5 py-0.5 rounded text-[10px] font-bold bg-red-100 text-red-800">ORIGEN</span>'
                    : '';

                li.innerHTML = `
                    <div class="flex items-start gap-3 overflow-hidden w-full">
                        <div class="drag-handle text-slate-300 hover:text-slate-500 mt-1 cursor-grab active:cursor-grabbing flex-shrink-0" title="Arrastrar para reordenar">
                            <svg xmlns="http://www.w3.org/2000/svg" class="h-5 w-5" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M4 6h16M4 12h16M4 18h16" />
                            </svg>
                        </div>
                        <div class="flex-shrink-0 w-7 h-7 rounded-full ${index === 0 ? 'bg-red-600' : 'bg-emerald-600'} text-white flex items-center justify-center text-xs font-bold mt-0.5 shadow-inner">
                            ${index + 1}
                        </div>
                        <div class="flex-grow min-w-0 pr-2">
                            <p class="text-sm font-bold text-slate-800 truncate leading-tight">
                                ${stop.name} ${badge}
                            </p>
                            <p class="text-[11px] text-slate-500 line-clamp-2 mt-1 leading-snug" title="${stop.display_name}">${stop.display_name}</p>
                        </div>
                    </div>
                    <button onclick="removeStop(${index})" class="flex-shrink-0 text-slate-300 hover:text-red-500 hover:bg-red-50 p-1.5 rounded transition-colors" title="Eliminar parada">

        async function calculateRoute(optimize) {
            if (stops.length < 2) return;

            loadingOverlay.classList.remove('hidden');
            loadingOverlay.querySelector('p').textContent = optimize 
                ? "Calculando ruta óptima (TSP)..." 
                : "Calculando ruta en orden manual...";
                
            statsPanel.classList.remove('scale-100');
            statsPanel.classList.add('scale-0');

            const coordsString = stops.map(s => `${s.lon},${s.lat}`).join(';');
            
            // Si optimize es true usamos API Trip (reordena), si es false usamos API Route (respeta tu orden)
            let osrmUrl = '';
            if (optimize) {
                osrmUrl = `https://router.project-osrm.org/trip/v1/driving/${coordsString}?roundtrip=false&source=first&destination=any&geometries=geojson&overview=full`;
            } else {
                osrmUrl = `https://router.project-osrm.org/route/v1/driving/${coordsString}?geometries=geojson&overview=full`;
            }

            try {
                const response = await fetch(osrmUrl);
                if (!response.ok) throw new Error("Error HTTP " + response.status);
                
                const data = await response.json();

                if (data.code === "Ok") {
                    let distance, duration, geometry;

                    if (optimize) {
                        // 1. Reordenar nuestro array de paradas según la optimización automática de OSRM
                        const newOrder = new Array(stops.length);
                        data.waypoints.forEach((wp, originalIndex) => {
                            newOrder[wp.waypoint_index] = stops[originalIndex];
                        });
                        stops = newOrder;
                        
                        distance = data.trips[0].distance;
                        duration = data.trips[0].duration;
                        geometry = data.trips[0].geometry;
                    } else {
                        // En modo manual NO reordenamos el array, usamos la geometría devuelta tal cual
                        distance = data.routes[0].distance;
                        duration = data.routes[0].duration;
                        geometry = data.routes[0].geometry;
                    }

                    // 2. Dibujar la línea en el mapa
                    drawRouteOnMap(geometry);

                    // 3. Actualizar la Interfaz
                    refreshMarkerIcons();
                    updateUI();
                    displayStats(distance, duration);

                    showToast(optimize ? "Ruta óptima calculada con éxito." : "Ruta calculada en el orden seleccionado.", "success");
                } else {
                    throw new Error(data.message || "No se pudo calcular la ruta.");
                }

            } catch (error) {
                console.error("Error en OSRM:", error);
                showToast("Error calculando la ruta. Puede que los puntos estén muy lejos o haya un problema de conexión.", "error");
            } finally {
                loadingOverlay.classList.add('hidden');
            }
        }

        function drawRouteOnMap(geoJsonGeometry) {
            clearRoute(); // Borrar línea anterior si la hay
            
            // Crear una polilínea Leaflet a partir del GeoJSON
            currentRouteLayer = L.geoJSON(geoJsonGeometry, {
                style: {
                    color: '#059669', // Verde esmeralda
                    weight: 6,
                    opacity: 0.8,
                    lineCap: 'round',
                    lineJoin: 'round',
                    dashArray: '10, 10', // Estilo de línea punteada/discontinua para darle toque técnico
                    className: 'route-path-animated' // Para futura animación CSS si se desea
                }
            }).addTo(map);

            // Ajustar el zoom para que se vea toda la ruta
            map.fitBounds(currentRouteLayer.getBounds(), { padding: [40, 40] });
        }

        function clearRoute() {
            if (currentRouteLayer) {
                map.removeLayer(currentRouteLayer);
                currentRouteLayer = null;
            }
        }

        function displayStats(distanceMeters, driveTimeSeconds) {
            // El origen no cuenta como entrega, por eso (stops.length - 1)
            const deliveryCount = Math.max(0, stops.length - 1); 
            const totalDropTimeSeconds = deliveryCount * SECONDS_PER_DELIVERY;
            const totalOverallTimeSeconds = driveTimeSeconds + totalDropTimeSeconds;

            // Llenar el HTML
            document.getElementById('statDistance').textContent = (distanceMeters / 1000).toFixed(1) + " km";
            document.getElementById('statDriveTime').textContent = formatSecondsToHM(driveTimeSeconds);
            document.getElementById('statDropCount').textContent = deliveryCount;
            document.getElementById('statDropTime').textContent = formatSecondsToHM(totalDropTimeSeconds);
            document.getElementById('statTotalTime').textContent = formatSecondsToHM(totalOverallTimeSeconds);

            // Mostrar el panel con animación
            statsPanel.classList.remove('hidden');
            // Timeout pequeño para permitir que el 'hidden' desaparezca antes de escalar (animación CSS)
            setTimeout(() => {
                statsPanel.classList.remove('scale-0');
                statsPanel.classList.add('scale-100');
            }, 50);
        }

        function formatSecondsToHM(seconds) {
            const h = Math.floor(seconds / 3600);
            const m = Math.floor((seconds % 3600) / 60);
            if (h > 0) return `${h}h ${m}m`;
            if (m > 0) return `${m} min`;
            return "< 1 min";
        }

            let icon = `<svg class="w-5 h-5 text-blue-400" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M13 16h-1v-4h-1m1-4h.01M21 12a9 9 0 11-18 0 9 9 0 0118 0z"></path></svg>`;
            
            if(type === 'error') {
                bg = 'bg-red-800';
                icon = `<svg class="w-5 h-5 text-red-400" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M12 8v4m0 4h.01M21 12a9 9 0 11-18 0 9 9 0 0118 0z"></path></svg>`;
            } else if (type === 'success') {
                bg = 'bg-emerald-800';
                icon = `<svg class="w-5 h-5 text-emerald-400" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M9 12l2 2 4-4m6 2a9 9 0 11-18 0 9 9 0 0118 0z"></path></svg>`;

                icon = `<svg class="w-5 h-5 text-amber-300" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M12 9v2m0 4h.01m-6.938 4h13.856c1.54 0 2.502-1.667 1.732-3L13.732 4c-.77-1.333-2.694-1.333-3.464 0L3.34 16c-.77 1.333.192 3 1.732 3z"></path></svg>`;
            }

            toast.className = `${bg} text-white px-4 py-3 rounded shadow-lg flex items-center gap-3 transform transition-all duration-300 translate-y-10 opacity-0`;

            }, 4000);
        }

        new Sortable(listElement, {
            animation: 150,
            handle: '.drag-handle', // Solo se mueve si tocas el icono de las barras
            ghostClass: 'sortable-ghost',
            onEnd: function (evt) {
                if (evt.oldIndex === evt.newIndex) return;

                // Modificar nuestro Array de estado interno
                const movedItem = stops.splice(evt.oldIndex, 1)[0];
                stops.splice(evt.newIndex, 0, movedItem);

                // Como el orden cambió, la ruta antigua ya no sirve
                clearRoute();
                statsPanel.classList.remove('scale-100');
                statsPanel.classList.add('scale-0');

                // Refrescar mapa y repintar lista para actualizar los números
                refreshMarkerIcons();
                updateUI();
            }
        });

        exportGoogleMapsBtn.addEventListener('click', () => {
            if (stops.length < 2) return;

            // Obtener el origen (primera parada) y destino (última parada)
            const origin = `${stops[0].lat},${stops[0].lon}`;
            const destination = `${stops[stops.length - 1].lat},${stops[stops.length - 1].lon}`;
            
            // Construir la base de la URL de navegación (api=1 asegura compatibilidad en móviles y PC)
            let mapsUrl = `https://www.google.com/maps/dir/?api=1&origin=${origin}&destination=${destination}&travelmode=driving`;

            // Si hay puntos intermedios (waypoints), los agregamos a la URL separados por el carácter "|"
            if (stops.length > 2) {
                const waypoints = stops.slice(1, stops.length - 1)
                                       .map(stop => `${stop.lat},${stop.lon}`)
                                       .join('|');
                
                // Nota: Google Maps suele tener un límite de entre 9 y 10 waypoints en la URL. 
                // Si la ruta tiene más, Google podría truncarla en la app móvil.
                mapsUrl += `&waypoints=${waypoints}`;
                
                if (stops.length > 11) {
                    showToast("Google Maps limita las paradas por enlace. Mostrando las primeras 10.", "warning");
                }
            }

            // Abrir la URL en una nueva pestaña (en móvil esto suele forzar que se abra la App de Maps)
            window.open(mapsUrl, '_blank');
        });
